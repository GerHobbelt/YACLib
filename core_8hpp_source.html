112 template <typename T>
113 void Call(Core& self, T&& r) noexcept {
114 try {
115 constexpr bool kIsRun = std::is_same_v<T, Unit>;
116 static_assert(kIsRun || std::is_same_v<T, Result<Arg, E>>);
117 if constexpr (kIsRun || is_invocable_v<Invoke, Result<Arg, E>>) {
118 CallResolveAsync(self, std::forward<T>(r));
119 } else {
120 CallResolveState(self, std::forward<T>(r));
121 }
122 } catch (...) {
123 self.Set(std::current_exception());
124 }
125 }
126
127 private:
128 void CallResolveState(Core& self, Result<Arg, E>&& r) {
129 const auto state = r.State();
130 if constexpr (is_invocable_v<Invoke, Arg>) {
131 if (state == ResultState::Value) {
132 CallResolveAsync(self, std::move(r).Value());
133 } else if (state == ResultState::Exception) {
134 self.Set(std::move(r).Exception());
135 } else {
136 YACLIB_DEBUG(state != ResultState::Error, "state should be Error, but here it's Empty");
137 self.Set(std::move(r).Error());
138 }
139 } else {
140 /**
141 * We can't use this strategy for other Func::Invoke,
142 * because in that case user will not have compile error for that case:
143 * MakeFuture<void>() // state == ResultState::Value
144 * .ThenInline([](E/std::exception_ptr) -> yaclib::Result/Future<double> {
145 * throw std::runtime_error{""};
146 * }).ThenInline([](yaclib::Result<double>) { // need double value, we only have void
147 * return 1;
148 * });
149 * TLDR: Before and after the "recovery" callback must have the same value type
150 */
151 constexpr bool kIsException = is_invocable_v<Invoke, std::exception_ptr>;
152 constexpr bool kIsError = is_invocable_v<Invoke, E>;
153 static_assert(kIsException ^ kIsError, "Recovery callback should be invokable with std::exception_ptr or E");
154 constexpr auto kState = kIsException ? ResultState::Exception : ResultState::Error;
155 if (state == kState) {
156 using T = std::conditional_t<kIsException, std::exception_ptr, E>;
157 CallResolveAsync(self, std::move(r).template Extract<T>());
158 } else {
159 self.Set(std::move(r));
160 }
161 }
162 }

164 template <typename T>
165 void CallResolveAsync(Core& self, T&& value) {
166 if constexpr (Async) {
167 auto future = CallResolveVoid(std::forward<T>(value));
168 auto core = std::exchange(future.GetCore(), nullptr);
169 self.IncRef();
170 core->SetCallbackInline(self, true);
171 } else {
172 self.Set(CallResolveVoid(std::forward<T>(value)));
173 }
174 }

176 template <typename T>
177 auto CallResolveVoid(T&& value) {
178 constexpr bool kArgVoid = std::is_same_v<T, Unit>;
179 constexpr bool kRetVoid = std::is_void_v<invoke_t<Invoke, std::conditional_t<kArgVoid, void, T>>>;
180 if constexpr (kRetVoid) {
181 if constexpr (kArgVoid) {
182 std::forward<Invoke>(_func)();
183 } else {
184 std::forward<Invoke>(_func)(std::forward<T>(value));
185 }
186 return Unit{};
187 } else if constexpr (kArgVoid) {
188 return std::forward<Invoke>(_func)();
189 } else {
190 return std::forward<Invoke>(_func)(std::forward<T>(value));
191 }
192 }
