188/**
189 * Function for create Ready Future
190 *
191 * \tparam V if not default value, it's type of Future value
192 * \tparam E type of Future error, by default its
193 * \tparam Args if single, and V default, then used as type of Future value
194 * \param args for fulfill Future
195 * \return Ready Future
196 */
197template <typename V = Unit, typename E = StopError, typename... Args>
198auto MakeFuture(Args&&... args) {
199 if constexpr (sizeof...(Args) == 0) {
200 return Future{MakeIntrusive<detail::ResultCore<void, E>>(Unit{})};
201 } else if constexpr (sizeof...(Args) == 1) {
202 using T = std::conditional_t<std::is_same_v<V, Unit>, head_t<Args...>, V>;
203 static_assert(!std::is_same_v<T, Unit>);
204 return Future{MakeIntrusive<detail::ResultCore<T, E>>(std::forward<Args>(args)...)};
205 } else {
206 static_assert(!std::is_same_v<V, Unit>);
207 return Future{MakeIntrusive<detail::ResultCore<V, E>>(std::forward<Args>(args)...)};
208 }
209}
