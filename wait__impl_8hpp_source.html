17template <typename Event, typename Timeout, typename Range>
18bool WaitRange(const Timeout& timeout, const Range& range, std::size_t count) {
19 AtomicCounter<Event, SetOneDeleter> event{count + 1};
20 // event ref counter = n + 1, it is optimization: we don't want to notify when return true immediately
21 auto const wait_count = range([&](BaseCore& core) {
22 return core.Empty() && core.SetWait(event);
23 });
24 if (wait_count == 0 || event.SubEqual(count - wait_count + 1)) {
25 return true;
26 }
27 auto token = event.Make();
28 std::size_t reset_count = 0;
29 if constexpr (!std::is_same_v<Timeout, NoTimeoutTag>) {
30 // If you have problem with TSAN here, check this link: https://github.com/google/sanitizers/issues/1259
31 // TLDR: new pthread function is not supported by thread sanitizer yet.
32 if (event.Wait(token, timeout)) {
33 return true;
34 }
35 reset_count = range([](BaseCore& core) {
36 return core.ResetWait();
37 });
38 if (reset_count != 0 && (reset_count == wait_count || event.SubEqual(reset_count))) {
39 return false;
40 }
41 // We know we have `wait_count - reset_count` Results, but we must wait until event was not used by cores
42 }
43 event.Wait(token);
44 return reset_count == 0;
45}

47template <typename Event, typename Timeout, typename... Cores>
48bool WaitCore(const Timeout& timeout, Cores&... cores) {
49 static_assert(sizeof...(cores) >= 1, "Number of futures must be at least one");
50 static_assert((... && std::is_same_v<BaseCore, Cores>), "Futures must be Future in Wait function");
51 auto range = [&](auto&& func) {
52 return (... + static_cast<std::size_t>(func(cores)));
53 };
54 return WaitRange<Event>(timeout, range, sizeof...(cores));
55}

57template <typename Event, typename Timeout, typename Iterator>
58bool WaitIterator(const Timeout& timeout, Iterator it, std::size_t count) {
59 static_assert(is_future_base_v<typename std::iterator_traits<Iterator>::value_type>,
60 "Wait function Iterator must be point to some FutureBase");
61 if (count == 0) {
62 return true;
63 }
64 auto range = [&](auto&& func) {
65 std::size_t wait_count = 0;
66 std::conditional_t<std::is_same_v<Timeout, NoTimeoutTag>, Iterator&, Iterator> range_it = it;
67 for (std::size_t i = 0; i != count; ++i) {
68 wait_count += static_cast<std::size_t>(func(*range_it->GetCore()));
69 ++range_it;
70 }
71 return wait_count;
72 };
73 return WaitRange<Event>(timeout, range, count);
74}
